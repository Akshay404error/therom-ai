def factorial (n : Nat) : Nat :=
  match n with
  | 0 => 1
  | n + 1 => (n + 1) * factorial n

theorem factorial_positive (n : Nat) : 
  factorial n > 0 := by
  induction n with
  | zero => simp [factorial]
  | succ n ih => 
    simp [factorial]
    apply Nat.mul_pos
    · exact Nat.succ_pos n
    · exact ih
